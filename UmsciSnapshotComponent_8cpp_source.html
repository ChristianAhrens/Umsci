118void UmsciSnapshotComponent::paint(juce::Graphics& g)
119{
120 const auto bounds = getLocalBounds();
121 const int contentWidth = getWidth() - getGrabStripWidth();
122 const auto contentBounds = bounds.withWidth(contentWidth);
123 const auto stripBounds = bounds.withLeft(contentWidth);
124
125 // ── Panel background ──────────────────────────────────────────────────────
126 g.setColour(findColour(juce::ResizableWindow::backgroundColourId).withAlpha(0.93f));
127 g.fillRoundedRectangle(bounds.toFloat().reduced(1.0f), 5.0f);
128
129 // ── Border in highlight colour ─────────────────────────────────────────────
130 g.setColour(m_highlightColour);
131 g.drawRoundedRectangle(bounds.toFloat().reduced(1.0f), 5.0f, 1.5f);
132
133 // ── Vertical separator between content and grab strip ─────────────────────
134 g.setColour(m_highlightColour.withAlpha(0.35f));
135 g.drawVerticalLine(contentWidth, 5.0f, static_cast<float>(getHeight()) - 5.0f);
136
137 // ── Content ───────────────────────────────────────────────────────────────
138 paintContent(g, contentBounds);
139
140 // ── Grab strip ────────────────────────────────────────────────────────────
141 paintGrabStrip(g, stripBounds);
142}

159void UmsciSnapshotComponent::paintContent(juce::Graphics& g, juce::Rectangle<int> contentBounds)
160{
161 const int pad = getPadding();
162 const int titleRowH = getTitleRowHeight();
163 const int rowH = getContentRowHeight();
164 const float titleFont = getTitleFontSize();
165 const float bodyFont = getContentFontSize();
166
167 auto inner = contentBounds.reduced(pad, pad);
168
169 g.setColour(findColour(juce::Label::textColourId));
170 g.setFont(juce::Font(juce::FontOptions(titleFont, juce::Font::bold)));
171 g.drawText("Upmix indicator snapshot", inner.removeFromTop(titleRowH), juce::Justification::centredLeft, true);
172
173 if (m_snapshotData.has_value())
174 {
175 const auto& s = *m_snapshotData;
176 g.setFont(juce::Font(juce::FontOptions(bodyFont, juce::Font::plain)));
177 g.drawText("Rotation: " + juce::String(s.rot, 3), inner.removeFromTop(rowH), juce::Justification::centredLeft, true);
178 g.drawText("Scale: " + juce::String(s.scale, 3), inner.removeFromTop(rowH), juce::Justification::centredLeft, true);
179 g.drawText("H.Scale: " + juce::String(s.heightScale, 3), inner.removeFromTop(rowH), juce::Justification::centredLeft, true);
180 g.drawText("A.Stretch: " + juce::String(s.angleStretch, 3), inner.removeFromTop(rowH), juce::Justification::centredLeft, true);
181 g.drawText("Offset: " + juce::String(s.offsetX, 3) +
182 ", " + juce::String(s.offsetY, 3), inner.removeFromTop(rowH), juce::Justification::centredLeft, true);
183 }
184 else
185 {
186 g.setColour(findColour(juce::Label::textColourId).withAlpha(0.45f));
187 g.setFont(juce::Font(juce::FontOptions(bodyFont, juce::Font::plain)));
188 g.drawText("No snapshot stored", inner, juce::Justification::centred, true);
189 }
190}

192void UmsciSnapshotComponent::paintGrabStrip(juce::Graphics& g, juce::Rectangle<int> stripBounds)
193{
194 constexpr float pillW = 4.0f;
195 constexpr float pillH = 2.0f;
196 constexpr float gap = 5.0f;
197
198 const float cx = static_cast<float>(stripBounds.getCentreX());
199 const float cy = static_cast<float>(stripBounds.getCentreY());
200
201 g.setColour(m_highlightColour.withAlpha(0.75f));
202 for (int i = -1; i <= 1; ++i)
203 {
204 const auto pill = juce::Rectangle<float>(
205 cx - pillW * 0.5f,
206 cy + static_cast<float>(i) * gap - pillH * 0.5f,
207 pillW, pillH);
208 g.fillRoundedRectangle(pill, pillH * 0.5f);
209 }
210}

212void UmsciSnapshotComponent::mouseUp(const juce::MouseEvent& e)
213{
214 if (m_recallButton->getBounds().contains(e.getPosition()))
215 {
216 if (onRecallRequested) onRecallRequested();
217 return;
218 }
219
220 if (m_storeButton->getBounds().contains(e.getPosition()))
221 {
222 if (onStoreRequested) onStoreRequested();
223 return;
224 }
225
226 if (e.x >= (getWidth() - getGrabStripWidth()) && onStateChangeRequested)
227 {
228 const auto newState = (m_state == PanelState::Tucked) ? PanelState::Visible : PanelState::Tucked;
229 onStateChangeRequested(newState);
230 }
231}

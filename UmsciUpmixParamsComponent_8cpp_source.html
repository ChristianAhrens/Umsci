21UmsciUpmixParamsComponent::UmsciUpmixParamsComponent()
22{
23 setOpaque(false);
24 setInterceptsMouseClicks(true, true);
25
26 m_spreadSlider = std::make_unique<juce::Slider>(juce::Slider::LinearHorizontal,
27 juce::Slider::NoTextBox);
28 m_spreadSlider->setRange(0.0, 1.0, 0.0);
29 m_spreadSlider->setValue(static_cast<double>(m_spread), juce::dontSendNotification);
30 m_spreadSlider->setColour(juce::Slider::thumbColourId, m_highlightColour);
31 m_spreadSlider->setColour(juce::Slider::trackColourId, m_highlightColour.withAlpha(0.5f));
32 m_spreadSlider->onValueChange = [this] {
33 if (m_suppressCallbacks) return;
34 m_spread = static_cast<float>(m_spreadSlider->getValue());
35 checkMismatch();
36 if (onSpreadChanged) onSpreadChanged(m_spread);
37 };
38 addAndMakeVisible(m_spreadSlider.get());
39
40 auto initButton = [this](std::unique_ptr<juce::TextButton>& btn, const juce::String& label) {
41 btn = std::make_unique<juce::TextButton>(label);
42 btn->setClickingTogglesState(false);
43 btn->setColour(juce::TextButton::buttonColourId, juce::Colours::transparentBlack);
44 btn->setColour(juce::TextButton::buttonOnColourId, m_highlightColour);
45 addAndMakeVisible(btn.get());
46 };
47 initButton(m_delayModeOff, "Off");
48 initButton(m_delayModeTight, "Tight");
49 initButton(m_delayModeFull, "Full");
50
51 m_delayModeOff->onClick = [this] {
52 if (m_suppressCallbacks) return;
53 m_delayMode = 0;
54 updateDelayModeButtonStates();
55 checkMismatch();
56 if (onDelayModeChanged) onDelayModeChanged(0);
57 };
58 m_delayModeTight->onClick = [this] {
59 if (m_suppressCallbacks) return;
60 m_delayMode = 1;
61 updateDelayModeButtonStates();
62 checkMismatch();
63 if (onDelayModeChanged) onDelayModeChanged(1);
64 };
65 m_delayModeFull->onClick = [this] {
66 if (m_suppressCallbacks) return;
67 m_delayMode = 2;
68 updateDelayModeButtonStates();
69 checkMismatch();
70 if (onDelayModeChanged) onDelayModeChanged(2);
71 };
72 updateDelayModeButtonStates();
73
74 if (auto xml = juce::XmlDocument::parse(BinaryData::sync_problem_24dp_svg))
75 {
76 m_syncProblemDrawable = juce::Drawable::createFromSVG(*xml);
77 m_syncProblemDrawable->replaceColour(juce::Colours::black, m_highlightColour);
78 }
79}

260void UmsciUpmixParamsComponent::resized()
261{
262 const int pad = getPadding();
263 const int titleH = getTitleRowHeight();
264 const int rowH = getContentRowHeight();
265 const int gap = rowH / 4;
266 const int contentW = getWidth() - getGrabStripWidth();
267 const int innerW = contentW - 2 * pad;
268 const int innerX = pad;
269 const int labelW = innerW / 4;
270
271 // Row 1: "Spread" label (painted) + slider — gap below title
272 m_spreadSlider->setBounds(innerX + labelW, pad + titleH + gap, innerW - labelW, rowH);
273
274 // Row 2: "Delay:" label (painted) + three equal-width buttons — gap below row 1
275 const int btnAreaW = innerW - labelW;
276 const int btnW = btnAreaW / 3;
277 const int btnX = innerX + labelW;
278 const int btnY = pad + titleH + gap + rowH + gap;
279 m_delayModeOff->setBounds(btnX, btnY, btnW, rowH);
280 m_delayModeTight->setBounds(btnX + btnW, btnY, btnW, rowH);
281 m_delayModeFull->setBounds(btnX + 2 * btnW, btnY, btnAreaW - 2 * btnW, rowH);
282}

284void UmsciUpmixParamsComponent::paintContent(juce::Graphics& g, juce::Rectangle<int> contentBounds)
285{
286 const int pad = getPadding();
287 const int titleH = getTitleRowHeight();
288 const int rowH = getContentRowHeight();
289 const int gap = rowH / 4;
290 const float titleFont = getTitleFontSize();
291 const float bodyFont = getContentFontSize();
292
293 auto inner = contentBounds.reduced(pad, pad);
294 const int labelW = inner.getWidth() / 4;
295
296 g.setColour(findColour(juce::Label::textColourId));
297 g.setFont(juce::Font(juce::FontOptions(titleFont, juce::Font::bold)));
298 g.drawText("Upmix ch. parameters", inner.removeFromTop(titleH), juce::Justification::centredLeft, true);
299
300 inner.removeFromTop(gap);
301
302 g.setFont(juce::Font(juce::FontOptions(bodyFont, juce::Font::plain)));
303
304 auto spreadRow = inner.removeFromTop(rowH);
305 g.drawText("Spread", spreadRow.removeFromLeft(labelW), juce::Justification::centredLeft, true);
306
307 inner.removeFromTop(gap);
308
309 auto delayRow = inner.removeFromTop(rowH);
310 g.drawText("Delay:", delayRow.removeFromLeft(labelW), juce::Justification::centredLeft, true);

313void UmsciUpmixParamsComponent::paintGrabStrip(juce::Graphics& g, juce::Rectangle<int> stripBounds)
314{
315 constexpr float pillW = 4.0f;
316 constexpr float pillH = 2.0f;
317 constexpr float gap = 5.0f;
318
319 const float cx = static_cast<float>(stripBounds.getCentreX());
320 const float cy = static_cast<float>(stripBounds.getCentreY());
321
322 g.setColour(m_highlightColour.withAlpha(0.75f));
323 for (int i = -1; i <= 1; ++i)
324 {
325 const auto pill = juce::Rectangle<float>(
326 cx - pillW * 0.5f,
327 cy + static_cast<float>(i) * gap - pillH * 0.5f,
328 pillW, pillH);
329 g.fillRoundedRectangle(pill, pillH * 0.5f);
330 }
331}

333void UmsciUpmixParamsComponent::mouseUp(const juce::MouseEvent& e)
334{
335 const int grabX = getWidth() - getGrabStripWidth();
336
337 if (e.x >= grabX)
338 {
339 // Grab strip toggles panel state (always reachable).
340 if (onStateChangeRequested)
341 {
342 const auto newState = (m_state == PanelState::Tucked) ? PanelState::Visible : PanelState::Tucked;
343 onStateChangeRequested(newState);
344 }
345 }
346 else if (isTimerRunning())
347 {
348 // Content area click while out of sync: push current aggregate values to device.
349 // Children are blocked (setInterceptsMouseClicks false) so every click lands here.
350 if (onSpreadChanged) onSpreadChanged(m_spread);
351 if (onDelayModeChanged) onDelayModeChanged(m_delayMode);
352 }
353}

364void UmsciUpmixParamsComponent::checkMismatch()
365{
366 constexpr float kSpreadTol = 0.01f;
367 bool mismatch = false;
368
369 for (auto id : m_monitoredSourceIds)
370 {
371 auto spreadIt = m_deviceSpread.find(id);
372 if (spreadIt == m_deviceSpread.end()
373 || std::abs(spreadIt->second - m_spread) > kSpreadTol)
374 {
375 mismatch = true;
376 break;
377 }
378 auto delayIt = m_deviceDelayMode.find(id);
379 if (delayIt == m_deviceDelayMode.end()
380 || static_cast<int>(delayIt->second) != m_delayMode)
381 {
382 mismatch = true;
383 break;
384 }
385 }
386
387 if (mismatch && !isTimerRunning())
388 {
389 startTimer(500);
390 setChildrenMouseInterception(false);
391 }
392 else if (!mismatch && isTimerRunning())
393 {
394 stopTimer();
395 m_flashState = false;
396 setAlpha(1.0f);
397 setChildrenMouseInterception(true);
398 repaint();
399 }
400}

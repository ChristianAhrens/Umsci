47DeviceController::DeviceController()
48{
49 // Pre-build definition maps before the connection is opened.
50 CreateKnownONosMap();
51
52 m_ocp1IPAddress = juce::IPAddress("127.0.0.1");
53 m_ocp1Port = 50014;
54 m_ocp1Timeout = 100;
55
56 // false = callbacks on socket thread, not message thread.
57 m_ocp1Connection = std::make_unique<NanoOcp1::NanoOcp1Client>(m_ocp1IPAddress.toString(), m_ocp1Port, false);
58
59 m_ocp1Connection->onConnectionEstablished = [=]() {
60 stopTimer(); // Connection succeeded — no more retries needed.
61
62 // Reset device-model state: the GUID may differ if the device was replaced.
63 m_ocp1DeviceGUID = "";
64 m_ocp1DeviceStackIdent = -1;
65 m_connectedDbDeviceModel = DbDeviceModel::InvalidDev;
66
67 // Query the GUID first. Subscriptions follow in ProcessGuidAndSubscribe()
68 // once we know which OCA revision the firmware supports.
69 QueryObjectValue(RemoteObject::Fixed_GUID, {});
70 };
71
72 m_ocp1Connection->onConnectionLost = [=]() {
73 DeleteObjectSubscriptions();
74 ClearPendingHandles();
75
76 m_ocp1DeviceGUID = "";
77 m_ocp1DeviceStackIdent = -1;
78 m_connectedDbDeviceModel = DbDeviceModel::InvalidDev;
79
80 if (getState() > State::Connecting) // lost a live connection — try to reconnect
81 {
82 postMessage(new StateChangeMessage(State::Connecting));
83 startTimer(m_ocp1Timeout * 20); // timer fires timerCallback() → connectToSocket()
84 }
85 };
86
87 m_ocp1Connection->onDataReceived = [=](const NanoOcp1::ByteVector& data) {
88 return ocp1MessageReceived(data);
89 };
90}

101void DeviceController::setState(const State& s, juce::NotificationType notificationType)
102{
103 std::function<juce::String(State)> stateToString = [=](State state) {
104 switch (state)
105 {
106 case Disconnected:
107 return "disco";
108 case Connecting:
109 return "cntng";
110 case Subscribing:
111 return "sbscrbng";
112 case Subscribed:
113 return "sbscrbd";
114 case GetValues:
115 return "gtvls";
116 case Connected:
117 return "cnctd";
118 default:
119 return "ERR";
120 }
121 };
122
123 if (m_currentState != s)
124 {
125 DBG(juce::String(__FUNCTION__) << " " << stateToString(s));
126 m_currentState = s;
127
128 if (s == State::GetValues)
129 startTimer(m_ocp1Timeout * 20); // (re)start timeout; fires retryPendingGetValues() if device drops a response
130 else if (s == State::Connected)
131 stopTimer();
132
133 if (onStateChanged && juce::NotificationType::sendNotification == notificationType)
134 onStateChanged(s);
135 }
136}

143bool DeviceController::connect()
144{
145 if (State::Disconnected != getState())
146 {
147 DBG(juce::String(__FUNCTION__) << " - nothing to do as we're not disconnected");
148 return false;
149 }
150 DBG(__FUNCTION__);
151
152 setState(State::Connecting);
153
154 // prepare to restart connection attempt after some large timeout, in case something got stuck...
155 startTimer(m_ocp1Timeout * 20);
156
157 timerCallback(); // avoid codeclones by manually trigger the timed connection attempt once
158
159 return true;
160}

165

167
168 if (m_ocp1Connection)
169 m_ocp1Connection->disconnect(m_ocp1Timeout);
170
171 stopTimer();

194void DeviceController::timerCallback()
195{
196 if (State::Connecting == getState())
197 {
198 m_ocp1Connection->connectToSocket(m_ocp1IPAddress.toString(), m_ocp1Port, m_ocp1Timeout);
199 }
200 else if (State::GetValues == getState())
201 {
202 retryPendingGetValues();
203 }
204 else
205 {
206 jassertfalse;
207 disconnect();
208 }

211void DeviceController::retryPendingGetValues()
212{
213 std::vector<std::uint32_t> staleONos;
214 {
215 std::lock_guard<std::mutex> l(m_pendingHandlesMutex);
216 staleONos.reserve(m_pendingGetValueHandlesWithONo.size());
217 for (auto const& [handle, ono] : m_pendingGetValueHandlesWithONo)
218 staleONos.push_back(ono);
219 m_pendingGetValueHandlesWithONo.clear();
220 }
221
222 DBG(juce::String(__FUNCTION__) << " re-querying " << staleONos.size() << " unanswered getvalue ONos");
223
224 for (auto ono : staleONos)
225 {
226 auto it = m_ONoToROIMap.find(ono);
227 if (it != m_ONoToROIMap.end())
228 {
229 auto const& [roi, addr] = it->second;
230 QueryObjectValue(roi, addr);
231 }
232 }

578bool DeviceController::ocp1MessageReceived(const NanoOcp1::ByteVector& data)
579{
580 std::unique_ptr<NanoOcp1::Ocp1Message> msgObj = NanoOcp1::Ocp1Message::UnmarshalOcp1Message(data);
581 if (msgObj)
582 {
583 switch (msgObj->GetMessageType())
584 {
585 case NanoOcp1::Ocp1Message::Notification:
586 {
587 NanoOcp1::Ocp1Notification* notifObj = static_cast<NanoOcp1::Ocp1Notification*>(msgObj.get());
588
589 if (UpdateObjectValue(notifObj))
590 return true;
591
592 DBG(juce::String(__FUNCTION__) << " Got an unhandled OCA notification for ONo 0x"
593 << juce::String::toHexString(notifObj->GetEmitterOno()));
594 return false;
595 }
596 case NanoOcp1::Ocp1Message::Response:
597 {
598 NanoOcp1::Ocp1Response* responseObj = static_cast<NanoOcp1::Ocp1Response*>(msgObj.get());
599
600 auto handle = responseObj->GetResponseHandle();
601 if (responseObj->GetResponseStatus() != 0)
602 {
603 DBG(juce::String(__FUNCTION__) << " Got an OCA response (handle:" << NanoOcp1::HandleToString(handle) <<
604 ") with status " << NanoOcp1::StatusToString(responseObj->GetResponseStatus()));
605
606 auto externalId = -1;
607 PopPendingSubscriptionHandle(handle);
608 PopPendingGetValueHandle(handle);
609 PopPendingSetValueHandle(handle, externalId);
610
611 return false;
612 }
613 else if (PopPendingSubscriptionHandle(handle))
614 {
615 if (!HasPendingSubscriptions())
616 {
617 // All subscriptions were confirmed
618 //DBG(juce::String(__FUNCTION__) << " All NanoOcp1 subscriptions were confirmed (handle:"
619 // << NanoOcp1::HandleToString(handle) << ")");
620
621 postMessage(new StateChangeMessage(State::Subscribed));
622 if (HasPendingGetValues())
623 postMessage(new StateChangeMessage(State::GetValues));
624 else
625 postMessage(new StateChangeMessage(State::Connected));
626 }
627 return true;
628 }
629 else
630 {
631 auto GetValONo = PopPendingGetValueHandle(handle);
632 if (0x00 != GetValONo)
633 {
634 if (!UpdateObjectValue(GetValONo, responseObj))
635 {
636 DBG(juce::String(__FUNCTION__) << " Got an unhandled OCA getvalue response message (handle:"
637 << NanoOcp1::HandleToString(handle) + ", targetONo:0x" << juce::String::toHexString(GetValONo) << ")");
638 return false;
639 }
640 else
641 {
642 if (!HasPendingGetValues())
643 {
644 // All getvalues were confirmed
645 //DBG(juce::String(__FUNCTION__) << " All pending NanoOcp1 getvalue commands were confirmed (handle:"
646 // << NanoOcp1::HandleToString(handle) << ")");
647
648 if (!HasPendingSubscriptions())
649 postMessage(new StateChangeMessage(State::Connected));
650 else
651 postMessage(new StateChangeMessage(State::Subscribing));
652 }
653 return true;
654 }
655 }
656
657 auto externalId = -1;
658 auto SetValONo = PopPendingSetValueHandle(handle, externalId);
659 if (0x00 != SetValONo)
660 {
661 if (!HasPendingSetValues())
662 {
663 // All subscriptions were confirmed
664 //DBG(juce::String(__FUNCTION__) << " All pending NanoOcp1 setvalue commands were confirmed (handle:"
665 // << NanoOcp1::HandleToString(handle) << ")");
666 }
667 return true;
668 }
669
670 DBG(juce::String(__FUNCTION__) << " Got an OCA response for UNKNOWN handle " << NanoOcp1::HandleToString(handle) <<
671 "; status " << NanoOcp1::StatusToString(responseObj->GetResponseStatus()) <<
672 "; paramCount " << juce::String(responseObj->GetParamCount()));
673
674 return false;
675 }
676 }
677 case NanoOcp1::Ocp1Message::KeepAlive:
678 {
679 jassertfalse; //todo
681 //if (m_messageListener)
682 //{
683 // m_messageListener->OnProtocolMessageReceived(this, RemoteObject::HeartbeatPong, RemoteObjectMessageData());
684 // return true;
685 //}
686 //else
687 // return false;
688 }
689 default:
690 break;
691 }
692 }
693
694 return false;
695}

1152bool DeviceController::SetOcaRevisionAndDeviceModel(const juce::String& guid)
1153{
1154 if (guid.length() != 8) // d&b GUIDs are always exactly 8 characters
1155 return false;
1156 if (!guid.startsWith("DB00")) // mandatory d&b manufacturer prefix
1157 return false;
1158
1159 int deviceBytes = 2;
1160 DbDeviceModel dbDeviceModel; // last two characters decide the model
1161 if (guid.getLastCharacters(deviceBytes) == "D0") // DS100
1162 dbDeviceModel = DbDeviceModel::DS100;
1163 else if (guid.getLastCharacters(deviceBytes) == "D1") // DS100D
1164 dbDeviceModel = DbDeviceModel::DS100D;
1165 else if (guid.getLastCharacters(deviceBytes) == "D2") // DS100M
1166 dbDeviceModel = DbDeviceModel::DS100M;
1167 else
1168 return false;
1169
1170 int versionBytesIndexStart = 4; // in the Guid the version is after "DB00" and has two bytes/characters
1171 auto versionChars = guid.substring(versionBytesIndexStart, versionBytesIndexStart + 2); // will get two characters
1172 int ocp1DeviceStackIdent;
1173 switch (dbDeviceModel)
1174 {
1175 case DbDeviceModel::DS100:
1176 if (versionChars >= "0C") // DS100 added scalability with FW version "0C"
1177 ocp1DeviceStackIdent = 1;
1178 else
1179 ocp1DeviceStackIdent = 0;
1180 DBG(juce::String(__FUNCTION__) << " detected DS100 (ocp stack:" << ocp1DeviceStackIdent << ") " << guid);
1181 break;
1182 case DbDeviceModel::DS100D:
1183 ocp1DeviceStackIdent = 1; // this was implemented pre-release of DS100D and assuming there will be no FW-version without scalability
1184 DBG(juce::String(__FUNCTION__) << " detected DS100D (ocp stack:" << ocp1DeviceStackIdent << ") " << guid);
1185 break;
1186 case DbDeviceModel::DS100M:
1187 if (versionChars >= "02") // DS100M added scalability with FW version "02"
1188 ocp1DeviceStackIdent = 1;
1189 else
1190 ocp1DeviceStackIdent = 0;
1191 DBG(juce::String(__FUNCTION__) << " detected DS100M (ocp stack:" << ocp1DeviceStackIdent << ") " << guid);
1192 break;
1193 default:
1194 return false;
1195
1196 }
1197 m_connectedDbDeviceModel = dbDeviceModel;
1198 m_ocp1DeviceStackIdent = ocp1DeviceStackIdent;
1199 return true;
1200}

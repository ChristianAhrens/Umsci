79 static UpmixSnapshot fromString(const juce::String& s)
80 {
81 UpmixSnapshot p;
82 for (auto& token : juce::StringArray::fromTokens(s, ";", ""))
83 {
84 auto kv = juce::StringArray::fromTokens(token.trim(), "=", "");
85 if (kv.size() != 2) continue;
86 auto key = kv[0].trim();
87 auto val = kv[1].trim().getFloatValue();
88 if (key == "rot") p.rot = val;
89 else if (key == "scale") p.scale = val;
90 else if (key == "heightScale") p.heightScale = val;
91 else if (key == "angleStretch") p.angleStretch = val;
92 else if (key == "offsetX") p.offsetX = val;
93 else if (key == "offsetY") p.offsetY = val;
94 }
95 return p;
96 }

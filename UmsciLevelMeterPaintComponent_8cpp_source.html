35UmsciLevelMeterPaintComponent::MeterRing UmsciLevelMeterPaintComponent::buildMeterRing(
36 juce::Point<float> centre, float meterRadius,
37 const std::vector<ChannelPosition>& positions) const
38{
39 MeterRing ring;
40 ring.meterRadius = meterRadius;
41 for (auto const& pos : positions)
42 {
43 juce::Point<float> dir = pos.screenPos - centre;
44 float dirLen = dir.getDistanceFromOrigin();
45 if (dirLen < 1.0f)
46 continue;
47 MeterRing::Channel ch;
48 ch.sourceId = pos.sourceId;
49 ch.maxPoint = centre + (dir / dirLen) * meterRadius;
50 ring.channels.push_back(ch);
51 }
52 return ring;
53}

55void UmsciLevelMeterPaintComponent::setRingPositions(
56 juce::Point<float> centre,
57 float subCircleRadius,
58 const std::vector<ChannelPosition>& floorPositions,
59 const std::vector<ChannelPosition>& heightPositions)
60{
61 m_centre = centre;
62
63 // The meter circle for each ring sits just inside the indicator sub-circles:
64 // radius = min(distance from centre to any dot) - subCircleRadius.
65 // In circle mode all dots are equidistant; in rectangle mode this picks the
66 // closest edge-centre dots, giving a consistent circular meter shape.
67 auto computeMeterRadius = [&](const std::vector<ChannelPosition>& positions) -> float
68 {
69 float minDist = std::numeric_limits<float>::max();
70 for (auto const& pos : positions)
71 minDist = std::min(minDist, pos.screenPos.getDistanceFrom(centre));
72 return std::max(0.0f, minDist - subCircleRadius);
73 };
74
75 // Channels arrive in bed order (L, R, C, Ls, Rs, …); sort by screen angle so the
76 // polygon connects them in clockwise order and never self-intersects.
77 auto sortByAngle = [&](std::vector<ChannelPosition> positions) -> std::vector<ChannelPosition>
78 {
79 std::sort(positions.begin(), positions.end(), [&](const ChannelPosition& a, const ChannelPosition& b)
80 {
81 return std::atan2(a.screenPos.y - centre.y, a.screenPos.x - centre.x)
82 < std::atan2(b.screenPos.y - centre.y, b.screenPos.x - centre.x);
83 });
84 return positions;
85 };
86
87 m_floorRing = buildMeterRing(centre, computeMeterRadius(floorPositions), sortByAngle(floorPositions));
88 m_heightRing = buildMeterRing(centre, computeMeterRadius(heightPositions), sortByAngle(heightPositions));
89 m_hasGeometry = !m_floorRing.channels.empty();
90
91 repaint();
92}

94void UmsciLevelMeterPaintComponent::setLevelValue(std::int16_t sourceId, float normalizedLevel)
95{
96 normalizedLevel = juce::jlimit(0.0f, 1.0f, normalizedLevel);
97 m_currentLevel[sourceId] = normalizedLevel;
98
99 auto& hold = m_holdLevel[sourceId];
100 if (normalizedLevel >= hold)
101 {
102 hold = normalizedLevel;
103 m_holdTimestampMs[sourceId] = juce::Time::getMillisecondCounterHiRes();
104 }
105
106 m_levelDirty.store(true, std::memory_order_relaxed);
107}

110void UmsciLevelMeterPaintComponent::timerCallback()
111{
112 auto nowMs = juce::Time::getMillisecondCounterHiRes();
113 bool holdChanged = false;
114 for (auto& [sourceId, holdTimestamp] : m_holdTimestampMs)
115 {
116 auto it = m_holdLevel.find(sourceId);
117 if (it != m_holdLevel.end() && it->second > 0.0f
118 && nowMs - holdTimestamp > static_cast<double>(kHoldDurationMs))
119 {
120 it->second = 0.0f;
121 holdChanged = true;
122 }
123 }
124
125 if (m_levelDirty.exchange(false, std::memory_order_relaxed) || holdChanged)
126 repaint();
127}

140void UmsciLevelMeterPaintComponent::paintMeterRing(juce::Graphics& g, const MeterRing& ring) const
141{
142 if (ring.channels.empty() || ring.meterRadius < 1.0f)
143 return;
144
145 auto rmsColour = getLookAndFeel()
146 .findColour(JUCEAppBasics::CustomLookAndFeel::ColourIds::MeteringRmsColourId);
147 auto holdColour = getLookAndFeel()
148 .findColour(JUCEAppBasics::CustomLookAndFeel::ColourIds::MeteringHoldColourId);
149
150 // --- 1. Reference circle (always visible) ---
151 g.setColour(rmsColour.withAlpha(0.40f));
152 g.drawEllipse(m_centre.x - ring.meterRadius, m_centre.y - ring.meterRadius,
153 ring.meterRadius * 2.0f, ring.meterRadius * 2.0f, 1.0f);
154
155 // Pre-compute per-channel vectors:
156 // centerToMax = centre − maxPoint (points from edge back to centre)
157 // widthOffset = 90° rotation of unit-dir × halfMeterWidth (perpendicular bar width)
158 struct ChannelVecs
159 {
160 juce::Point<float> centerToMax;
161 juce::Point<float> widthOffset;
162 bool valid = false;
163 };
164 std::vector<ChannelVecs> vecs(ring.channels.size());
165
166 for (size_t i = 0; i < ring.channels.size(); ++i)
167 {
168 juce::Point<float> dir = ring.channels[i].maxPoint - m_centre;
169 float len = dir.getDistanceFromOrigin();
170 if (len < 1.0f)
171 continue;
172 dir /= len;
173 vecs[i].centerToMax = m_centre - ring.channels[i].maxPoint;
174 vecs[i].widthOffset = { -dir.y * kHalfMeterWidth, dir.x * kHalfMeterWidth };
175 vecs[i].valid = true;
176 }
177
178 // --- 2. Static spoke lines: centre → maxPoint (channel direction markers) ---
179 g.setColour(rmsColour.withAlpha(0.35f));
180 for (size_t i = 0; i < ring.channels.size(); ++i)
181 {
182 if (!vecs[i].valid)
183 continue;
184 g.drawLine(juce::Line<float>(m_centre, ring.channels[i].maxPoint), 1.0f);
185 }
186
187 // --- 3. Level polygon (filled) at current level ---
188 {
189 juce::Path levelPath;
190 bool started = false;
191 for (size_t i = 0; i < ring.channels.size(); ++i)
192 {
193 if (!vecs[i].valid)
194 continue;
195
196 float level = 0.0f;
197 auto it = m_currentLevel.find(ring.channels[i].sourceId);
198 if (it != m_currentLevel.end())
199 level = it->second;
200
201 auto levelPoint = m_centre - (vecs[i].centerToMax * level);
202
203 if (!started)
204 {
205 levelPath.startNewSubPath(levelPoint - vecs[i].widthOffset);
206 started = true;
207 }
208 else
209 {
210 levelPath.lineTo(levelPoint - vecs[i].widthOffset);
211 }
212 levelPath.lineTo(levelPoint + vecs[i].widthOffset);
213 }
214 levelPath.closeSubPath();
215
216 g.setColour(rmsColour.withAlpha(0.35f));
217 g.fillPath(levelPath);
218 }
219
220 // --- 4. Per-channel spoke lines (isolated channels only) ---
221 // When both angular neighbours are silent the polygon is degenerate at this channel,
222 // so draw a full spoke (centre → levelPoint) to keep the reading visible.
223 // When at least one neighbour is active the polygon already covers the area.
224 {
225 const size_t N = ring.channels.size();
226 auto neighbourHasLevel = [&](size_t idx) -> bool {
227 auto nit = m_currentLevel.find(ring.channels[idx].sourceId);
228 return nit != m_currentLevel.end() && nit->second > 0.001f;
229 };
230
231 g.setColour(rmsColour.withAlpha(0.75f));
232 for (size_t i = 0; i < N; ++i)
233 {
234 if (!vecs[i].valid)
235 continue;
236
237 float level = 0.0f;
238 auto it = m_currentLevel.find(ring.channels[i].sourceId);
239 if (it != m_currentLevel.end())
240 level = it->second;
241
242 if (level < 0.001f)
243 continue;
244
245 if (neighbourHasLevel((i + N - 1) % N) || neighbourHasLevel((i + 1) % N))
246 continue;
247
248 auto levelPoint = m_centre - (vecs[i].centerToMax * level);
249 g.drawLine(juce::Line<float>(m_centre, levelPoint), kMeterLineWidth);
250 }
251 }
252
253 // --- 5. Hold: stroked outline polygon or isolated cap ---
254 // Contiguous runs of hold-active channels are each drawn as a separate stroked path.
255 // A full-ring run (all channels active) is closed into a polygon; partial runs are
256 // left open so no diagonal closing segment crosses the silent gap.
257 // Isolated channels (both angular neighbours silent) get a perpendicular cap tick.
258 {
259 const size_t N = ring.channels.size();
260
261 auto holdOf = [&](size_t idx) -> float {
262 auto hit = m_holdLevel.find(ring.channels[idx].sourceId);
263 return (hit != m_holdLevel.end()) ? hit->second : 0.0f;
264 };
265
266 g.setColour(holdColour.darker(0.3f).withAlpha(0.90f));
267
268 // Find the first channel after a gap so wrap-around runs are not split.
269 size_t startIdx = 0;
270 for (size_t k = 0; k < N; ++k)
271 {
272 if (holdOf(k) > 0.01f && holdOf((k + N - 1) % N) <= 0.01f)
273 {
274 startIdx = k;
275 break;
276 }
277 }
278
279 juce::Path holdPath;
280 bool inRun = false;
281 int runChannelCount = 0;
282
283 auto flushRun = [&]()
284 {
285 if (inRun)
286 {
287 if (runChannelCount == static_cast<int>(N))
288 holdPath.closeSubPath();
289 else
290 {
291 // Partial run: route the closing edge through the origin so the
292 // polygon closes cleanly at the centre rather than leaving an open path.
293 holdPath.lineTo(m_centre);
294 holdPath.closeSubPath();
295 }
296 g.strokePath(holdPath, juce::PathStrokeType(1.5f));
297 holdPath.clear();
298 inRun = false;
299 runChannelCount = 0;
300 }
301 };
302
303 for (size_t ki = 0; ki < N; ++ki)
304 {
305 const size_t i = (startIdx + ki) % N;
306 if (!vecs[i].valid) { flushRun(); continue; }
307
308 const float hold = holdOf(i);
309 const bool prevHasHold = holdOf((i + N - 1) % N) > 0.01f;
310 const bool nextHasHold = holdOf((i + 1) % N) > 0.01f;
311
312 if (hold < 0.01f) { flushRun(); continue; }
313
314 if (!prevHasHold && !nextHasHold)
315 {
316 // Isolated: perpendicular cap tick
317 flushRun();
318 auto holdPoint = m_centre - (vecs[i].centerToMax * hold);
319 g.drawLine(juce::Line<float>(
320 holdPoint - vecs[i].widthOffset,
321 holdPoint + vecs[i].widthOffset), 1.5f);
322 continue;
323 }
324
325 auto holdPoint = m_centre - (vecs[i].centerToMax * hold);
326 if (!inRun)
327 {
328 holdPath.startNewSubPath(holdPoint - vecs[i].widthOffset);
329 inRun = true;
330 }
331 else
332 {
333 holdPath.lineTo(holdPoint - vecs[i].widthOffset);
334 }
335 holdPath.lineTo(holdPoint + vecs[i].widthOffset);
336 ++runChannelCount;
337 }
338 flushRun();
339 }

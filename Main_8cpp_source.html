73 /*
74 This class implements the desktop window that contains an instance of
75 our MainComponent class.
76 */
77 class MainWindow : public juce::DocumentWindow, juce::DarkModeSettingListener
78 {
79 public:
80 MainWindow(const juce::String& name, const juce::String& commandLine) : juce::DocumentWindow(name,
81 juce::Desktop::getInstance().getDefaultLookAndFeel()
82 .findColour(juce::ResizableWindow::backgroundColourId),
83 juce::DocumentWindow::allButtons)
84 {
85 ignoreUnused(commandLine);
86
87 setUsingNativeTitleBar(true);
88 auto mainComponent = std::make_unique<MainComponent>();
89 mainComponent->onPaletteStyleChange = [=](int paletteStyle, bool followLocalStyle) {
90 m_followLocalStyle = followLocalStyle;
91 applyPaletteStyle(static_cast<JUCEAppBasics::CustomLookAndFeel::PaletteStyle>(paletteStyle));
92 };
93 mainComponent->onSetFullscreenWindow = [=](bool fullscreenWindow) { setFullscreenWindow(fullscreenWindow); };
94 setContentOwned(mainComponent.release(), true);
95
96#if JUCE_IOS || JUCE_ANDROID
97 setFullScreen(true);
98 juce::Desktop::getInstance().setScreenSaverEnabled(false);
99#elif JUCE_LINUX
100 juce::Desktop::getInstance().setKioskModeComponent(getTopLevelComponent(), false);
101#else
102 setResizable(true, true);
103 setResizeLimits(400, 350, 30000, 30000);
104 centreWithSize(getWidth(), getHeight());
105#endif
106
107 setVisible(true);
108
109 juce::Desktop::getInstance().addDarkModeSettingListener(this);
110 darkModeSettingChanged(); // initially trigger correct colourscheme
111
112 // use the settings menu item call infrastructure to activate dark mode per default
113 if (auto mc = dynamic_cast<MainComponent*>(getContentComponent()))
114 mc->applySettingsOption(MainComponent::UmsciSettingsOption::LookAndFeel_Dark);
115
116 JUCEAppBasics::AppConfigurationBase::getInstance()->triggerWatcherUpdate();
117 }
118
119 void closeButtonPressed() override
120 {
121 juce::JUCEApplication::getInstance()->systemRequestedQuit();
122 }
123
124 void darkModeSettingChanged() override
125 {
126 if (!m_followLocalStyle)
127 return;
128
129 if (juce::Desktop::getInstance().isDarkModeActive())
130 {
131 // go dark
132 applyPaletteStyle(JUCEAppBasics::CustomLookAndFeel::PS_Dark);
133 }
134 else
135 {
136 // go light
137 applyPaletteStyle(JUCEAppBasics::CustomLookAndFeel::PS_Light);
138 }
139
140 lookAndFeelChanged();
141 }
142
143 void applyPaletteStyle(JUCEAppBasics::CustomLookAndFeel::PaletteStyle paletteStyle)
144 {
145 m_lookAndFeel = std::make_unique<JUCEAppBasics::CustomLookAndFeel>(paletteStyle);
146 juce::Desktop::getInstance().setDefaultLookAndFeel(m_lookAndFeel.get());
147 }
148
149 void setFullscreenWindow(bool fullscreenWindow)
150 {
151#if JUCE_WINDOWS
152 if (fullscreenWindow)
153 juce::Desktop::getInstance().setKioskModeComponent(getTopLevelComponent(), false);
154 else
155 juce::Desktop::getInstance().setKioskModeComponent(nullptr, false);
156#elif JUCE_MAC
157 if (auto* topLevel = getTopLevelComponent())
158 {
159 if (auto* peer = topLevel->getPeer())
160 {
161 peer->setFullScreen(fullscreenWindow);
162 return;
163 }
164 }
165 jassertfalse;
166#endif
167 }
168
169 private:
170 std::unique_ptr<juce::LookAndFeel> m_lookAndFeel;
171 bool m_followLocalStyle = true;
172
173 JUCE_DECLARE_NON_COPYABLE_WITH_LEAK_DETECTOR(MainWindow)
174 };
175
176private:
177 std::unique_ptr<MainWindow> m_mainWindow;
178 std::unique_ptr<juce::TooltipWindow> m_toolTipWindowInstance;
179};

313void MainComponent::handleSettingsMenuResult(int selectedId)
314{
315 if (0 == selectedId)
316 return; // nothing selected, dismiss
317 else if (UmsciSettingsOption::LookAndFeel_First <= selectedId && UmsciSettingsOption::LookAndFeel_Last >= selectedId)
318 handleSettingsLookAndFeelMenuResult(selectedId);
319 else if (UmsciSettingsOption::ControlColour_First <= selectedId && UmsciSettingsOption::ControlColour_Last >= selectedId)
320 handleSettingsControlColourMenuResult(selectedId);
321 else if (UmsciSettingsOption::ConnectionSettings == selectedId)
322 showConnectionSettings();
323 else if (UmsciSettingsOption::FullscreenWindowMode == selectedId)
324 handleSettingsFullscreenModeToggleResult();
325 else if (UmsciSettingsOption::ControlFormat_First <= selectedId && UmsciSettingsOption::ControlFormat_Last >= selectedId)
326 handleSettingsControlFormatMenuResult(selectedId);
327 else if (UmsciSettingsOption::UpmixSettings == selectedId)
328 showUpmixSettings();
329 else if (UmsciSettingsOption::ControlSize_First <= selectedId && UmsciSettingsOption::ControlSize_Last >= selectedId)
330 handleSettingsControlSizeMenuResult(selectedId);
331 else
332 jassertfalse; // unhandled menu entry!?
333}

795void MainComponent::onConfigUpdated()
796{
797 // control config
798 auto controlConfigState = m_config->getConfigState(UmsciAppConfiguration::getTagName(UmsciAppConfiguration::TagID::CONTROLCONFIG));
799 if (controlConfigState && m_controlComponent)
800 m_controlComponent->setStateXml(controlConfigState.get());
801
802 // connection config
803 auto connectionConfigState = m_config->getConfigState(UmsciAppConfiguration::getTagName(UmsciAppConfiguration::TagID::CONNECTIONCONFIG));
804 if (connectionConfigState)
805 {
806 auto ocp1ConnectionEnabled = 1 == connectionConfigState->getIntAttribute(UmsciAppConfiguration::getAttributeName(UmsciAppConfiguration::AttributeID::ENABLED));
807 auto ocp1IP = juce::IPAddress(connectionConfigState->getStringAttribute(UmsciAppConfiguration::getAttributeName(UmsciAppConfiguration::AttributeID::IP)));
808 auto ocp1Port = connectionConfigState->getIntAttribute(UmsciAppConfiguration::getAttributeName(UmsciAppConfiguration::AttributeID::PORT));
809
810 if (m_connectingComponent)
811 m_connectingComponent->setConnectionParameters(ocp1IP, ocp1Port);
812
813 auto ocp1ConParams = DeviceController::getInstance()->getConnectionParameters();
814 if (std::get<0>(ocp1ConParams) != ocp1IP || std::get<1>(ocp1ConParams) != ocp1Port)
815 {
816 DeviceController::getInstance()->setConnectionParameters(ocp1IP, ocp1Port);
817 }
818
819 if (ocp1ConnectionEnabled != m_connectionToggleButton->getToggleState())
820 {
821 m_connectionToggleButton->setToggleState(ocp1ConnectionEnabled, juce::dontSendNotification);
822 if (ocp1ConnectionEnabled)
823 DeviceController::getInstance()->connect();
824 else
825 DeviceController::getInstance()->disconnect();
826 }
827 }
828
829 // visu config
830 auto visuConfigState = m_config->getConfigState(UmsciAppConfiguration::getTagName(UmsciAppConfiguration::TagID::VISUCONFIG));
831 if (visuConfigState)
832 {
833 auto lookAndFeelXmlElement = visuConfigState->getChildByName(UmsciAppConfiguration::getTagName(UmsciAppConfiguration::TagID::LOOKANDFEEL));
834 if (lookAndFeelXmlElement)
835 {
836 auto lookAndFeelSettingsOptionId = lookAndFeelXmlElement->getAllSubText().getIntValue();
837 handleSettingsLookAndFeelMenuResult(lookAndFeelSettingsOptionId);
838 }
839
840 auto controlColourXmlElement = visuConfigState->getChildByName(UmsciAppConfiguration::getTagName(UmsciAppConfiguration::TagID::CONTROLCOLOUR));
841 if (controlColourXmlElement)
842 {
843 auto controlColourSettingsOptionId = controlColourXmlElement->getAllSubText().getIntValue();
844 handleSettingsControlColourMenuResult(controlColourSettingsOptionId);
845 }
846
847 auto controlFormatXmlElement = visuConfigState->getChildByName(UmsciAppConfiguration::getTagName(UmsciAppConfiguration::TagID::CONTROLFORMAT));
848 if (controlFormatXmlElement)
849 {
850 auto controlFormatSettingsOptionId = controlFormatXmlElement->getAllSubText().getIntValue();
851 handleSettingsControlFormatMenuResult(controlFormatSettingsOptionId);
852 }
853
854 auto controlSizeXmlElement = visuConfigState->getChildByName(UmsciAppConfiguration::getTagName(UmsciAppConfiguration::TagID::CONTROLSIZE));
855 if (controlSizeXmlElement)
856 {
857 auto controlSizeSettingsOptionId = controlSizeXmlElement->getAllSubText().getIntValue();
858 handleSettingsControlSizeMenuResult(controlSizeSettingsOptionId);
859 }
860 }
861
862 // upmix config
863 auto upmixConfigState = m_config->getConfigState(
864 UmsciAppConfiguration::getTagName(UmsciAppConfiguration::TagID::UPMIXCONFIG));
865 if (upmixConfigState && m_controlComponent)
866 {
867 auto startId = upmixConfigState->getIntAttribute(
868 UmsciAppConfiguration::getAttributeName(UmsciAppConfiguration::AttributeID::UPMIXSOURCESTARTID), 1);
869 m_controlComponent->setUpmixSourceStartId(startId);
870 auto showAllSources = upmixConfigState->getIntAttribute(
871 UmsciAppConfiguration::getAttributeName(UmsciAppConfiguration::AttributeID::UPMIXSHOWALLSOURCES), 1) == 1;
872 m_controlComponent->setShowAllSources(showAllSources);
873 auto liveMode = upmixConfigState->getIntAttribute(
874 UmsciAppConfiguration::getAttributeName(UmsciAppConfiguration::AttributeID::UPMIXLIVEMODE), 0) == 1;
875 m_controlComponent->setUpmixLiveMode(liveMode);
876 auto upmixShape = UmsciUpmixIndicatorPaintNControlComponent::getShapeForName(
877 upmixConfigState->getStringAttribute(
878 UmsciAppConfiguration::getAttributeName(UmsciAppConfiguration::AttributeID::UPMIXSHAPE)));
879 m_controlComponent->setUpmixShape(upmixShape);
880 auto upmixRot = 0.0f;
881 if (auto* e = upmixConfigState->getChildByName(UmsciAppConfiguration::getTagName(UmsciAppConfiguration::TagID::UPMIXROT)))
882 upmixRot = e->getAllSubText().getFloatValue();
883 auto upmixScale = 1.0f;
884 if (auto* e = upmixConfigState->getChildByName(UmsciAppConfiguration::getTagName(UmsciAppConfiguration::TagID::UPMIXSCALE)))
885 upmixScale = e->getAllSubText().getFloatValue();
886 auto upmixHeightScale = 0.6f;
887 if (auto* e = upmixConfigState->getChildByName(UmsciAppConfiguration::getTagName(UmsciAppConfiguration::TagID::UPMIXHEIGHTSCALE)))
888 upmixHeightScale = e->getAllSubText().getFloatValue();
889 auto upmixAngleStretch = 1.0f;
890 if (auto* e = upmixConfigState->getChildByName(UmsciAppConfiguration::getTagName(UmsciAppConfiguration::TagID::UPMIXANGLESTRETCH)))
891 upmixAngleStretch = e->getAllSubText().getFloatValue();
892 m_controlComponent->setUpmixTransform(upmixRot, upmixScale, upmixHeightScale, upmixAngleStretch);
893 auto upmixOffsetX = 0.0f;
894 if (auto* e = upmixConfigState->getChildByName(UmsciAppConfiguration::getTagName(UmsciAppConfiguration::TagID::UPMIXOFFSETX)))
895 upmixOffsetX = e->getAllSubText().getFloatValue();
896 auto upmixOffsetY = 0.0f;
897 if (auto* e = upmixConfigState->getChildByName(UmsciAppConfiguration::getTagName(UmsciAppConfiguration::TagID::UPMIXOFFSETY)))
898 upmixOffsetY = e->getAllSubText().getFloatValue();
899 m_controlComponent->setUpmixOffset(upmixOffsetX, upmixOffsetY);
900 }

22UmsciZeroconfDiscoverComboComponent::UmsciZeroconfDiscoverComboComponent()
23{
24 m_comboBox = std::make_unique<juce::ComboBox>("ZeroconfServices");
25 m_comboBox->setTextWhenNothingSelected("Searching for OCA/OCP.1 devices...");
26 m_comboBox->setSelectedId(0, juce::dontSendNotification);
27 m_comboBox->setEnabled(false);
28 addAndMakeVisible(m_comboBox.get());
29
30 m_comboBox->onChange = [this]() {
31 auto idx = m_comboBox->getSelectedId() - 2; // id 1 = "searching" placeholder; services start at 2
32 if (idx >= 0 && idx < static_cast<int>(m_services.size()) && onServiceSelected)
33 onServiceSelected(m_services[static_cast<size_t>(idx)]);
34 };
35
36 m_searcher = std::make_unique<ZeroconfSearcher::ZeroconfSearcher>("OCA", "_oca._tcp");
37 m_searcher->AddListener(this);
38 m_searcher->StartSearching();
39}

57void UmsciZeroconfDiscoverComboComponent::updateComboBox()
58{
59 if (!m_searcher)
60 return;
61
62 m_services = m_searcher->GetServices();
63
64 m_comboBox->clear(juce::dontSendNotification);
65
66 if (m_services.empty())
67 {
68 m_comboBox->setTextWhenNothingSelected("Searching for OCA/OCP.1 devices...");
69 m_comboBox->setSelectedId(0, juce::dontSendNotification);
70 m_comboBox->setEnabled(false);
71 return;
72 }
73
74 m_comboBox->setEnabled(true);
75 m_comboBox->setTextWhenNothingSelected(juce::String(m_services.size()) + " OCA/OCP.1 devices available...");
76 for (int i = 0; i < static_cast<int>(m_services.size()); ++i)
77 {
78 auto& s = m_services[static_cast<size_t>(i)];
79 auto displayName = juce::String(s.name).upToFirstOccurrenceOf("._oca._tcp", false, true);
80 auto label = displayName + " (" + juce::String(s.ip) + ":" + juce::String(s.port) + ")";
81 m_comboBox->addItem(label, i + 2);
82 }
83 m_comboBox->setSelectedId(0, juce::dontSendNotification);
84}

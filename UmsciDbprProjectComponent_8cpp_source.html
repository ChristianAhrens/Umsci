123void UmsciDbprProjectComponent::paint(juce::Graphics& g)
124{
125 const auto bounds = getLocalBounds();
126 const int contentWidth = getWidth() - getGrabStripWidth();
127 const auto contentBounds = bounds.withWidth(contentWidth);
128 const auto stripBounds = bounds.withLeft(contentWidth);
129
130 // ── Panel background ──────────────────────────────────────────────────────
131 g.setColour(findColour(juce::ResizableWindow::backgroundColourId).withAlpha(0.93f));
132 g.fillRoundedRectangle(bounds.toFloat().reduced(1.0f), 5.0f);
133
134 // ── Border in highlight colour (dims during flash-off phase) ──────────────
135 const auto borderOpacity = (isTimerRunning() && !m_flashState) ? 0.25f : 1.0f;
136 g.setColour(m_highlightColour.withAlpha(borderOpacity));
137 g.drawRoundedRectangle(bounds.toFloat().reduced(1.0f), 5.0f, 1.5f);
138
139 // ── Vertical separator between content and grab strip ─────────────────────
140 g.setColour(m_highlightColour.withAlpha(0.35f));
141 g.drawVerticalLine(contentWidth, 5.0f, static_cast<float>(getHeight()) - 5.0f);
142
143 // ── Content ───────────────────────────────────────────────────────────────
144 paintContent(g, contentBounds);
145
146 // ── Grab strip ────────────────────────────────────────────────────────────
147 paintGrabStrip(g, stripBounds);
148}

169void UmsciDbprProjectComponent::paintContent(juce::Graphics& g, juce::Rectangle<int> contentBounds)
170{
171 const int pad = getPadding();
172 const int titleRowH = getTitleRowHeight();
173 const int rowH = getContentRowHeight();
174 const float titleFont = getTitleFontSize();
175 const float bodyFont = getContentFontSize();
176
177 auto inner = contentBounds.reduced(pad, pad);
178
179 if (m_hasData)
180 {
181 g.setColour(findColour(juce::Label::textColourId));
182 g.setFont(juce::Font(juce::FontOptions(titleFont, juce::Font::bold)));
183 g.drawText("DBPR Project", inner.removeFromTop(titleRowH), juce::Justification::centredLeft, true);
184
185 g.setFont(juce::Font(juce::FontOptions(bodyFont, juce::Font::plain)));
186
187 auto cmCount = 0;
188 for (auto const& kv : m_projectData.coordinateMappingData)
189 if (!kv.second.isNull()) ++cmCount;
190 auto spkCount = 0;
191 for (auto const& kv : m_projectData.speakerPositionData)
192 if (!kv.second.isNull()) ++spkCount;
193 auto soCount = 0;
194 for (auto const& kv : m_projectData.matrixInputData)
195 if (kv.second.isEnScene()) ++soCount;
196
197 const auto fgCount = static_cast<int>(m_projectData.functionGroupData.size());
198
199 const auto cmText = juce::String(cmCount) + " Coordinate Mapping" + (cmCount != 1 ? "s" : "");
200 const auto spkText = juce::String(spkCount) + " Speaker position" + (spkCount != 1 ? "s" : "");
201 const auto soText = juce::String(soCount) + " Soundobject" + (soCount != 1 ? "s" : "");
202 const auto fgText = juce::String(fgCount) + " Function group" + (fgCount != 1 ? "s" : "");
203
204 g.drawText(cmText, inner.removeFromTop(rowH), juce::Justification::centredLeft, true);
205 g.drawText(spkText, inner.removeFromTop(rowH), juce::Justification::centredLeft, true);
206 g.drawText(soText, inner.removeFromTop(rowH), juce::Justification::centredLeft, true);
207 g.drawText(fgText, inner.removeFromTop(rowH), juce::Justification::centredLeft, true);
208
209 // Sync-problem icon: centred over content, half the panel height, bright flash phase only
210 if (isTimerRunning() && m_flashState && m_syncProblemDrawable)
211 {
212 const int iconSize = getPanelHeight() / 2;
213 const auto iconBounds = juce::Rectangle<int>(contentBounds).withSizeKeepingCentre(iconSize, iconSize).toFloat();
214 m_syncProblemDrawable->drawWithin(g, iconBounds, juce::RectanglePlacement::centred, 1.0f);
215 }
216 }
217 else
218 {
219 g.setColour(findColour(juce::Label::textColourId).withAlpha(0.45f));
220 g.setFont(juce::Font(juce::FontOptions(bodyFont, juce::Font::plain)));
221 g.drawText("No project loaded", inner, juce::Justification::centred, true);
222 }
223}

225void UmsciDbprProjectComponent::paintGrabStrip(juce::Graphics& g, juce::Rectangle<int> stripBounds)
226{
227 // Three small rounded pill shapes centred in the grab strip, like a standard
228 // drag-handle indicator.
229 constexpr float pillW = 4.0f;
230 constexpr float pillH = 2.0f;
231 constexpr float gap = 5.0f;
232
233 const float cx = static_cast<float>(stripBounds.getCentreX());
234 const float cy = static_cast<float>(stripBounds.getCentreY());
235
236 g.setColour(m_highlightColour.withAlpha(0.75f));
237
238 for (int i = -1; i <= 1; ++i)
239 {
240 const auto pill = juce::Rectangle<float>(
241 cx - pillW * 0.5f,
242 cy + static_cast<float>(i) * gap - pillH * 0.5f,
243 pillW, pillH);
244 g.fillRoundedRectangle(pill, pillH * 0.5f);
245 }
246}

272void UmsciDbprProjectComponent::mouseUp(const juce::MouseEvent& e)
273{
274 // Buttons are click-transparent; the parent handles all interactions via
275 // coordinate checks to avoid any child/parent event-routing ambiguity.
276 if (m_deleteButton->getBounds().contains(e.getPosition()))
277 {
278 if (onDeleteRequested) onDeleteRequested();
279 return;
280 }
281
282 if (m_loadButton->getBounds().contains(e.getPosition()))
283 {
284 if (onLoadRequested) onLoadRequested();
285 return;
286 }
287
288 if (m_syncButton->getBounds().contains(e.getPosition()))
289 {
290 if (onSyncRequested) onSyncRequested();
291 return;
292 }
293
294 // Only the grab strip (rightmost getGrabStripWidth() pixels) triggers a state toggle.
295 if (e.x >= (getWidth() - getGrabStripWidth()) && onStateChangeRequested)
296 {
297 const auto newState = (m_state == PanelState::Tucked) ? PanelState::Visible : PanelState::Tucked;
298 onStateChangeRequested(newState);
299 }
300}
